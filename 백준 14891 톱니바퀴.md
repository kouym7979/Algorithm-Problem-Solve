## 백준 14891 톱니바퀴

```
#include <algorithm>
#include <vector>
#include <iostream>
#include <queue>
#include <math.h>
#include <string>
using namespace std;
int sum = 0;//최종 톱니바퀴의 점수 
vector<int>c[4];

void check_cycle(int r_num, int r_direct);
void turn_cycle(int t_num, int t_direct);

int main()
{

	int num, rotate, direct;//회전시킬 톱니바퀴 번호,회전 횟수, 회전 방향
	
	for (int i = 0; i < 4; i++)
	{
		string t = "";
		cin >> t;
		for (int j = 0; j < t.length(); j++)
		{
			int n = t[j] - '0';//문자열을 index단위로 정수형태로 저장하는 과정
			c[i].push_back(n);
		}
	}

	getchar();
	cin >> rotate;//회전 횟수
	for (int k = 0; k < rotate; k++)
	{
		cin >> num >> direct;//회전시킬 톱니바퀴 번호와 회전방향
		check_cycle(num-1, direct);
	}

	
	for (int i = 0; i < 4; i++)
	{
		if (c[i][0] == 0)//n극이면 0점
		{
			sum += 0;
		}
		else if (c[i][0] == 1)//s극이면 바퀴 순번의 제곱점수 
			sum += pow(2, i);
	}

	cout << sum << endl;
	return 0;
}

void turn_cycle(int t_num, int t_direct)
{
	int temp = 0;
	if (t_direct == 1)//시계방향
	{
		temp = c[t_num][7];
		for (int i = 7; i > 0; i--)
		{
			c[t_num][i] = c[t_num][i - 1];//시계방향으로 회전
		}
		c[t_num][0] = temp;
	}
	else //반시계 방향
	{
		temp = c[t_num][1];
		for (int i = 0; i < 7; i++)
		{
			c[t_num][i] = c[t_num][i + 1];
		}
		c[t_num][7] = temp;
	}



	return;
}

void check_cycle(int r_num, int r_direct)
{
	int x = r_direct;
	int y = r_direct * -1;
	if (r_num == 0)//회전 번호가 0번일 경우
	{
		//cout << "0번입니다" << endl;
		if (c[r_num][2] != c[r_num + 1][6])//서로 다른 극일경우
		{
			if (c[r_num + 1][2] != c[r_num + 2][6])
			{
				if (c[r_num + 2][2] != c[r_num + 2][6])
				{
					turn_cycle(r_num, x);//시계방향으로 회전하면 다음은 반시계
					turn_cycle(r_num + 1, y);
					turn_cycle(r_num + 2, x);
					turn_cycle(r_num + 3, y);
				}
				else
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num + 1, y);
					turn_cycle(r_num + 2, x);
				}
			}
			else
			{
				turn_cycle(r_num, x);
				turn_cycle(r_num + 1, y);
			}
		}
		else {
			turn_cycle(r_num, x);
		}
	}
	else if (r_num == 1)
	{
		//cout << "1번입니다" << endl;
		if (c[r_num][6] != c[r_num - 1][2])
		{
			if (c[r_num][2] == c[r_num + 1][6])
			{
				turn_cycle(r_num, x);
				turn_cycle(r_num-1, y);
			}
			else if (c[r_num][2] != c[r_num + 1][6])
			{
				if (c[r_num + 1][2] == c[r_num + 2][6])
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num - 1, y);
					turn_cycle(r_num + 1, y);
				}
				else
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num - 1, y);
					turn_cycle(r_num + 1, y);
					turn_cycle(r_num + 2, x);
				}
			}
		}
	}
	else if (r_num == 2)
	{
		//cout << "2번입니다" << endl;
		if (c[r_num][2] == c[r_num + 1][6])
		{
			if (c[r_num][6] != c[r_num - 1][2])
			{
				if (c[r_num - 1][6] != c[r_num - 2][2])
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num - 1, y);
					turn_cycle(r_num - 2, x);
				}
				else
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num - 1, y);
				}
			}
			else turn_cycle(r_num, x);
		}
		else
		{
			if (c[r_num][6] != c[r_num - 1][2])
			{
				if (c[r_num - 1][6] != c[r_num - 2][2])
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num + 1, y);
					turn_cycle(r_num - 1, y);
					turn_cycle(r_num - 2, x);
				}
				else
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num + 1, y);
					turn_cycle(r_num - 1, y);
				}
			}
			else {
				turn_cycle(r_num, x);
				turn_cycle(r_num + 1, y);
			}
		}
	}
	else if (r_num == 3)
	{
	//cout << "3번입니다" << endl;
		if (c[r_num][6] != c[r_num - 1][2])
		{
			if (c[r_num - 1][6] != c[r_num - 2][2])
			{
				if (c[r_num - 2][6] != c[r_num - 3][2])
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num - 1, y);
					turn_cycle(r_num - 2, x);
					turn_cycle(r_num - 3, y);
				}
				else
				{
					turn_cycle(r_num, x);
					turn_cycle(r_num - 1, y);
					turn_cycle(r_num - 2, x);
				}
			}
			else
			{
				turn_cycle(r_num, x);
				turn_cycle(r_num - 1, y);
			}
		}
		else 
			turn_cycle(r_num, x);
	}
	return;
}
```

___

문제설명: 톱니바퀴의 극을 나타내는 것을 문자열로 받아서 자릿수별로 정수형태로 저장하는 것과

톱니바퀴를 회전했을때의 각 톱니바퀴의 맞물리는 N극 S극 확인이 중요한 문제입니다.

코드는 벡터를 이용하여 톱니바퀴의 극을 비교하는 방식으로 구현했습니다.
