## 알려주는중

명강의



```
#include <string>
#include <vector>
#include <algorithm>
#include <iostream>
#include <queue>
using namespace std;

int solution(vector<int> scoville, int K) {
	int answer = 0;
	int degree = 0;//섞은 음식의 스코빌 지수
	int first, second;
	priority_queue<int, vector<int>, greater<int>> sco;//max_heap

	for (int i = 0; i < scoville.size(); i++)
	{
		sco.push(scoville[i]);//스코빌 지수들을 우선순위 큐에 넣어준다.
	}

	while (1)
	{
		if (sco.top() < K && sco.size()==1)
		{
			answer = -1;
			break;
		}
		
		else if (sco.top() > K) {
			break;
		}
		first = sco.top();
		sco.pop();
		second = sco.top();
		sco.pop();
		degree = first + second * 2;
		answer++;
		sco.push(degree);
	}

	

	return answer;
}
```
