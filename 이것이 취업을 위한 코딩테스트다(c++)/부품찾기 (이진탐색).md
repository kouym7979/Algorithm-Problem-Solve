## 부품찾기 (이진탐색)

```
#include <algorithm>
#include <iostream>
#include <vector>

//부품찾기
//전자 매장에는 N개의 부품이 있다.
//M개의 종류 부품을 주문할 때, M개의 부품 종류를 모두 확인해서 견적서를 작성해야한다.
//가게 안에 부품이 모두 있는지 확인하는 프로그램을 작성하라

//첫째줄에 N을 입력1<=N<=1000000
//둘째줄에 공백으로 구분하는 N개의 정수가 주어진다
//셋째줄에 정수 M이 주어지고 넷째줄에 M개의 정수가 주어진다
//각 부품이 존재하면 yes, 없으면 no를 출력한다.


using namespace std;
vector<int>a1;
vector<int>a2;
int n, m;

int main()
{
	ios_base::sync_with_stdio(NULL);
	bool check = false;
	int x;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> x;
		a1.push_back(x);
	}
	getchar();


	cin >> m;
	for (int i = 0;i < m; i++)
	{
		cin >> x;
		a2.push_back(x);
	}

	sort(a1.begin(), a1.end());//오름차순 정렬
	sort(a2.begin(), a2.end());

	
	for (int i = 0; i < m; i++)
	{
		check = false;
		int temp = a2[i];
		int start = 0;
		int end = a1.size() - 1;
		while (start<=end)
		{
			int mid = (start + end) / 2;

			if (a1[mid] == temp)
			{
				cout << " yes";
				check = true;
				break;
			}
			if (a1[mid] > temp)
			{
				end = mid-1;
			}
			else
				start = mid+1;
		}
		if (check == true) continue;
		else
			cout << " no";

	}

	return 0;
}
```
