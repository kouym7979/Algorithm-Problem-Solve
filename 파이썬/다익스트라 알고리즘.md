## 다익스트라 알고리즘

```
import sys
input=sys.stdin.readline

INF=int(1e9)

n,m=map(int,input().split())

start=int(input()) #시작노드

graph=[[] for i in range(n+1)] #각 노드에 연결되어 있는 노드에 대한 정보를 담는 리스트를 만들기

visited=[False]*(n+1) #방문한 적이 있는지 체크하는 목적의 리스트 만들기

distance= [INF]*(n+1) #최단 거리 테이블을 모두 무한으로 초기화

for _ in range(m): # 모든 간선 정보 입력받기
    a,b,c=map(int,input().split()) # a,b,c 각 시작, 끝, 가중치
    graph[a].append((b,c))


#방문하지 않은 노드 중에서, 가장 최단 거리가 짧은 노드의 번호를 반환
def get_smallest_node():
    min_value=INF
    index=0 #가장 최단 거리가 짧은 노드(인덱스)
    for i in range(1,n+1):
        if distance[i]<min_value and not visited[i]:
            min_value=distance[i]
            index=i
    return index


def dijkstra(start):
    distance[start]=0
    visited[start]=True
    for j in graph[start]:
        distance[j[0]]=j[1]

    #시작노드를 제외한 전체 n-1개의 노드에 대해 반복
    for i in range(n-1):
        now=get_smallest_node()
        visited[now]=True #현재 최단 거리가 가장 짧은 노드를 꺼내서, 방문처리
        for j in graph[now]:
            cost=distance[now]+j[1]
            #현재 노드를 거쳐서 다른 노드로 이동하는 거리가 더 짧은 경우
            if cost<distance[j[0]]:
                distance[j[0]]=cost

dijkstra(start)

for i in range(1,n+1): #도달할 수 없는 경우, 무한 출력
    if distance[i]==INF:
        print("INFINITY")
    else:
        print(distance[i])
```

출처: 이것이 취업을 위한 코딩테스트다



데이터입력이 많을 경우 sys.stdin.readline을 사용하는것이 좋다

그래프를 표현할 때는 노드의 번호를 인덱스로 하여 리스트에 바로 접근할 수 있도록 (노드의 개수+1)개로 할당하는 것이 유용하다.

위의 코드의 시간복잡도는 O(V^2)만큼 나온다.

따라서 노드의 개수 및 간선의 개수가 많을 때는 아래의 '개선된 다익스트라 알고리즘'을 사용하는것이 좋다

```
import heapq
import sys

input=sys.stdin.readline
INF=int(1e9)

n,m=map(int, input().split())

start=int(input())

graph=[[] for i in range(n+1)] #각 노드에 연결되어 있는 노드에 대한 정보를 담는 리스트

distanace=[INF]*(n+1) #최단 거리 테이블 초기화

for _ in range(m):
    a,b,c=map(int,input().split())
    graph[a].append((b,c))

def dijkstra(start):
    q=[]
    heapq.heappush(q,(0,start)) # 시작 노드로 가기 위한 최단 경로는 0으로 설정하여, 큐에 삽입
    distanace[start]=0
    while q: #큐가 비어있지 않다면
        #가장 최단 거리가 짧은 노드에 대한 정보 꺼내기
        dist,now=heapq.heappop(q)
        if distanace[now]<dist: #현재 노드가 이미 처리된 적이 있는 노드라면 무시
            continue
        #현재 노드와 연결된 다른 인접한 노드들을 확인
        for i in graph[now]:
            cost=dist+i[1]
            #현재 노드를 거쳐서, 다른 노드로 이동하는 거리가 더 짧은 경우
            if cost<distanace[i[0]]:
                distanace[i[0]]=cost
                heapq.heappush(q,(cost,i[0]))

dijkstra(start)

for i in range(1,n+1):
    if distanace[i]==INF:
        print("INFINITY")
    else:
        print(distanace[i])
```

개선된 알고리즘을 사용할 경우 시간 복잡도가 O(ElogV) 로 훨씬 빠르다. (E는 간선의 개수고 V는 노드의 개수이다)
